452constexpr size_t pad_size(size_t size)
453 {
454 // We emulate glibc here (values measured on Linux i386).
455 // FIXME: We should better copy the portable value definitions from glibc.
456 if ( size == 0 )
457 return 0; // glibc allocated 16 bytes anyway.
458
459 const int pad = 8;
460 if ( size < 12 )
461 return 2 * pad;
462
463 return ((size + 3) / pad + 1) * pad;
464 }

580template <typename T> std::vector<T> split(T s, const T& delim)
581 {
582 // If there's no delimiter, return a copy of the existing string.
583 if ( delim.empty() )
584 return {T(s)};
585
586 // If the delimiter won't fit in the string, just return a copy as well.
587 if ( s.size() < delim.size() )
588 return {T(s)};
589
590 std::vector<T> l;
591
592 const bool ends_in_delim = (s.substr(s.size() - delim.size()) == delim);
593
594 do
595 {
596 size_t p = s.find(delim);
597 l.push_back(s.substr(0, p));
598 if ( p == std::string::npos )
599 break;
600
601 s = s.substr(p + delim.size());
602 } while ( ! s.empty() );
603
604 if ( ends_in_delim )
605 l.emplace_back(T{});
606
607 return l;
608 }
