123// Returns the given table/set (which should be empty) coerced to
124// the given Zeek type. A separate function in order to deal with
125// error handling. Inlined because this gets invoked a lot.
126inline TableValPtr table_coerce__CPP(const ValPtr& v, const TypePtr& t)
127 {
128 TableVal* tv = v->AsTableVal();
129
130 if ( tv->Size() > 0 )
131 reporter->CPPRuntimeError("coercion of non-empty table/set");
132
133 return make_intrusive<TableVal>(cast_intrusive<TableType>(t), tv->GetAttrs());
134 }
