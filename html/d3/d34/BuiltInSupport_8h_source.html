9// Base class for tracking information about a single cat() argument, with
10// optimizations for some common cases.
11class CatArg
12 {
13public:
14 CatArg(std::string _s) : s(std::move(_s)) { max_size = s->size(); }
15
16 virtual ~CatArg() { }
17
18 size_t MaxSize(ZVal* zframe, int slot)
19 {
20 return max_size ? *max_size : ComputeMaxSize(zframe, slot);
21 }
22
23 virtual void RenderInto(ZVal* zframe, int slot, char*& res)
24 {
25 auto n = *max_size;
26 memcpy(res, s->data(), n);
27 res += n;
28 }
29
30protected:
31 CatArg() { }
32 CatArg(size_t _max_size) : max_size(_max_size) { }
33
34 virtual size_t ComputeMaxSize(ZVal* zframe, int slot) { return 0; }
35
36 // Present if max size is known a priori.
37 std::optional<size_t> max_size;
38
39 // Present if the argument is a constant.
40 std::optional<std::string> s;
41 };
