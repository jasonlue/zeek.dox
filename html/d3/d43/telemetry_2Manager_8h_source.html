447 template <class F> static void WithLabelNames(Span<const LabelView> xs, F continuation)
448 {
449 if ( xs.size() <= 10 )
450 {
451 std::string_view buf[10];
452 for ( size_t index = 0; index < xs.size(); ++index )
453 buf[index] = xs[index].first;
454
455 return continuation(Span{buf, xs.size()});
456 }
457 else
458 {
459 std::vector<std::string_view> buf;
460 for ( auto x : xs )
461 buf.emplace_back(x.first, x.second);
462
463 return continuation(Span{buf});
464 }
