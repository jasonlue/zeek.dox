147template <typename T> inline T Queue<T>::Get()
148 {
149 auto lock = acquire_lock(mutex[read_ptr]);
150
151 int old_read_ptr = read_ptr;
152
153 if ( messages[read_ptr].empty() &&
154 ! ((reader && reader->Killed()) || (writer && writer->Killed())) )
155 {
156 if ( has_data[read_ptr].wait_for(lock, std::chrono::seconds(5)) == std::cv_status::timeout )
157 return nullptr;
158 }
159
160 if ( messages[read_ptr].empty() )
161 return nullptr;
162
163 T data = messages[read_ptr].front();
164 messages[read_ptr].pop();
165
166 read_ptr = (read_ptr + 1) % NUM_QUEUES;
167 ++num_reads;
168
169 return data;
170 }
