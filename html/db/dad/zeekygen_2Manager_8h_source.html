240 bool Manager::IsUpToDate(const std::string& target_file, const std::vector<T*>& dependencies) const
241  {
242  struct stat s;
243 
244  if ( stat(target_file.c_str(), &s) < 0 )
245  {
246  if ( errno == ENOENT )
247  // Doesn't exist.
248  return false;
249 
250  reporter->InternalError("Zeekygen failed to stat target file '%s': %s", target_file.c_str(),
251  strerror(errno));
252  }
253 
254  if ( difftime(bro_mtime, s.st_mtime) > 0 )
255  return false;
256 
257  if ( difftime(config.GetModificationTime(), s.st_mtime) > 0 )
258  return false;
259 
260  for ( size_t i = 0; i < dependencies.size(); ++i )
261  if ( difftime(dependencies[i]->GetModificationTime(), s.st_mtime) > 0 )
262  return false;
263 
264  return true;
265  }
