130// Abstract class for creating a collection of initializers. T1 is
131// the type of the generated vector, T2 the type of its initializers.
132template <class T1, class T2> class CPP_AbstractInits
133 {
134public:
135 CPP_AbstractInits(std::vector<T1>& _inits_vec, int _offsets_set, std::vector<T2> _inits)
136 : inits_vec(_inits_vec), offsets_set(_offsets_set), inits(std::move(_inits))
137 {
138 // Compute how big to make the vector.
139 int num_inits = 0;
140
141 for ( const auto& cohort : inits )
142 num_inits += cohort.size();
143
144 inits_vec.resize(num_inits);
145 }
146
147 // Initialize the given cohort of elements.
148 void InitializeCohort(InitsManager* im, int cohort)
149 {
150 // Get this object's vector-of-vector-of-indices.
151 auto& offsets_vec = im->Indices(offsets_set);
152
153 if ( cohort == 0 )
154 DoPreInits(im, offsets_vec);
155
156 // Get the vector-of-indices for this cohort.
157 auto& cohort_offsets = im->Indices(offsets_vec[cohort]);
158
159 InitializeCohortWithOffsets(im, cohort, cohort_offsets);
160 }
161
162protected:
163 virtual void InitializeCohortWithOffsets(InitsManager* im, int cohort,
164 const std::vector<int>& cohort_offsets)
165 {
166 }
167
168 // Pre-initialize all elements requiring it.
169 virtual void DoPreInits(InitsManager* im, const std::vector<int>& offsets_vec) { }
170
171 // Generate a single element.
172 virtual void GenerateElement(InitsManager* im, T2& init, int offset) { }
173
174 // The initialization vector in its entirety.
175 std::vector<T1>& inits_vec;
176
177 // A meta-index for retrieving the vector-of-vector-of-indices.
178 int offsets_set;
179
180 // Indexed by cohort.
181 std::vector<T2> inits;
182 };

184// Manages an initialization vector that uses "custom" initializers
185// (tailored ones rather than initializers based on indexing).
186template <class T> using CPP_InitVec = std::vector<std::shared_ptr<CPP_Init<T>>>;
187template <class T> class CPP_CustomInits : public CPP_AbstractInits<T, CPP_InitVec<T>>
188 {
189public:
190 CPP_CustomInits(std::vector<T>& _inits_vec, int _offsets_set,
191 std::vector<CPP_InitVec<T>> _inits)
192 : CPP_AbstractInits<T, CPP_InitVec<T>>(_inits_vec, _offsets_set, std::move(_inits))
193 {
194 }
195
196private:
197 void DoPreInits(InitsManager* im, const std::vector<int>& offsets_vec) override
198 {
199 int cohort = 0;
200 for ( const auto& co : this->inits )
201 {
202 auto& cohort_offsets = im->Indices(offsets_vec[cohort]);
203 for ( auto i = 0U; i < co.size(); ++i )
204 co[i]->PreInit(im, this->inits_vec, cohort_offsets[i]);
205 ++cohort;
206 }
207 }
208
209 void InitializeCohortWithOffsets(InitsManager* im, int cohort,
210 const std::vector<int>& cohort_offsets) override
211 {
212 // Loop over the cohort's elements to initialize them.
213 auto& co = this->inits[cohort];
214 for ( auto i = 0U; i < co.size(); ++i )
215 co[i]->Generate(im, this->inits_vec, cohort_offsets[i]);
216 }
217 };

315// Abstract class for initializing basic (non-compound) constants. T1 is
316// the Zeek type for the constructed constant, T2 is the C++ type of its
317// initializer.
318//
319// In principle we could derive this from CPP_AbstractInits, though to do so
320// we'd need to convert the initializers to a vector-of-vector-of-T2, which
321// would trade complexity here for complexity in InitsInfo. So we instead
322// keep this class distinct, since at heart it's a simpler set of methods
323// and that way we can keep them as such here.
324template <class T1, typename T2> class CPP_AbstractBasicConsts
325 {
326public:
327 CPP_AbstractBasicConsts(std::vector<T1>& _inits_vec, int _offsets_set, std::vector<T2> _inits)
328 : inits_vec(_inits_vec), offsets_set(_offsets_set), inits(std::move(_inits))
329 {
330 inits_vec.resize(inits.size());
331 }
332
333 void InitializeCohort(InitsManager* im, int cohort)
334 {
335 ASSERT(cohort == 0);
336 auto& offsets_vec = im->Indices(offsets_set);
337 auto& cohort_offsets = im->Indices(offsets_vec[cohort]);
338 for ( auto i = 0U; i < inits.size(); ++i )
339 InitElem(im, cohort_offsets[i], i);
340 }
341
342protected:
343 virtual void InitElem(InitsManager* im, int offset, int index) { ASSERT(0); }
344
345protected:
346 // See CPP_AbstractInits for the nature of these.
347 std::vector<T1>& inits_vec;
348 int offsets_set;
349 std::vector<T2> inits;
350 };
