14 // Class for iterating over the elements of a table. Requires some care
15 // because the dictionary iterators need to be destructed when done.
16 
17 class TableIterInfo
18  {
19 public:
20  // No constructor needed, as all of our member variables are
21  // instead instantiated via BeginLoop(). This allows us to
22  // reuse TableIterInfo objects to lower the overhead associated
23  // with executing ZBody::DoExec for non-recursive functions.
24 
25  // We do, however, want to make sure that when we go out of scope,
26  // if we have any pending iterators we clear them.
27  ~TableIterInfo() { Clear(); }
28 
29  // Start looping over the elements of the given table. "_aux"
30  // provides information about the index variables, their types,
31  // and the type of the value variable (if any).
32  void BeginLoop(const TableVal* _tv, ZInstAux* _aux)
33  {
34  tv = _tv;
35  aux = _aux;
36  auto tvd = tv->AsTable();
37  tbl_iter = tvd->begin();
38  tbl_end = tvd->end();
39  }
40 
41  // True if we're done iterating, false if not.
42  bool IsDoneIterating() const { return *tbl_iter == *tbl_end; }
43 
44  // Indicates that the current iteration is finished.
45  void IterFinished() { ++*tbl_iter; }
46 
47  // Performs the next iteration (assuming IsDoneIterating() returned
48  // false), assigning to the index variables.
49  void NextIter(ZVal* frame)
50  {
51  auto ind_lv = tv->RecreateIndex(*(*tbl_iter)->GetHashKey());
52  for ( int i = 0; i < ind_lv->Length(); ++i )
53  {
54  ValPtr ind_lv_p = ind_lv->Idx(i);
55  auto& var = frame[aux->loop_vars[i]];
56  auto& t = aux->loop_var_types[i];
57  if ( ZVal::IsManagedType(t) )
58  ZVal::DeleteManagedType(var);
59  var = ZVal(ind_lv_p, t);
60  }
61 
62  IterFinished();
63  }

66  ZVal IterValue()

69  return ZVal(tev->GetVal(), aux->value_var_type);
70  }

73  void EndIter() { Clear(); }
