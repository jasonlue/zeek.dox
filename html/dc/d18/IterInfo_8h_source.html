14// Class for iterating over the elements of a table. Requires some care
15// because the dictionary iterators need to be destructed when done.
16
17class TableIterInfo
18 {
19public:
20 // No constructor needed, as all of our member variables are
21 // instead instantiated via BeginLoop(). This allows us to
22 // reuse TableIterInfo objects to lower the overhead associated
23 // with executing ZBody::DoExec for non-recursive functions.
24
25 // We do, however, want to make sure that when we go out of scope,
26 // if we have any pending iterators we clear them.
27 ~TableIterInfo() { Clear(); }
28
29 // Start looping over the elements of the given table. "_aux"
30 // provides information about the index variables, their types,
31 // and the type of the value variable (if any).
32 void BeginLoop(const TableVal* _tv, ZInstAux* _aux)
33 {
34 tv = _tv;
35 aux = _aux;
36 auto tvd = tv->AsTable();
37 tbl_iter = tvd->begin();
38 tbl_end = tvd->end();
39 }
40
41 // True if we're done iterating, false if not.
42 bool IsDoneIterating() const { return *tbl_iter == *tbl_end; }
43
44 // Indicates that the current iteration is finished.
45 void IterFinished() { ++*tbl_iter; }
46
47 // Performs the next iteration (assuming IsDoneIterating() returned
48 // false), assigning to the index variables.
49 void NextIter(ZVal* frame)
50 {
51 auto ind_lv = tv->RecreateIndex(*(*tbl_iter)->GetHashKey());
52 for ( int i = 0; i < ind_lv->Length(); ++i )
53 {
54 ValPtr ind_lv_p = ind_lv->Idx(i);
55 auto& var = frame[aux->loop_vars[i]];
56 auto& t = aux->loop_var_types[i];
57 if ( ZVal::IsManagedType(t) )
58 ZVal::DeleteManagedType(var);
59 var = ZVal(ind_lv_p, t);
60 }
61
62 IterFinished();
63 }
64
65 // For the current iteration, returns the corresponding value.
66 ZVal IterValue()
67 {
68 auto tev = (*tbl_iter)->value;
69 return ZVal(tev->GetVal(), aux->value_var_type);
70 }
71
72 // Called upon finishing the iteration.
73 void EndIter() { Clear(); }
74
75 // Called to explicitly clear any iteration state.
76 void Clear()
77 {
78 tbl_iter = std::nullopt;
79 tbl_end = std::nullopt;
80 }
81
82private:
83 // The table we're looping over. If we want to allow for the table
84 // going away before we're able to clear our iterators then we
85 // could change this to non-const and use Ref/Unref.
86 const TableVal* tv = nullptr;
87
88 // Associated auxiliary information.
89 ZInstAux* aux = nullptr;
90
91 std::optional<DictIterator<TableEntryVal>> tbl_iter;
92 std::optional<DictIterator<TableEntryVal>> tbl_end;
93 };

95// Class for simple step-wise iteration across an integer range.
96// Suitable for iterating over vectors or strings.

98class StepIterInfo
99 {
100public:
101 // We do some cycle-squeezing by not having a constructor to
102 // initialize our member variables, since we impose a discipline
103 // that any use of the object starts with InitLoop(). That lets
104 // us use quasi-static objects for non-recursive functions.
105
106 // Initializes for looping over the elements of a raw vector.
107 void InitLoop(const std::vector<std::optional<ZVal>>* _vv)
108 {
109 vv = _vv;
110 n = vv->size();
111 iter = 0;
112 }
113
114 // Initializes for looping over the elements of a raw string.
115 void InitLoop(const String* _s)
116 {
117 s = _s;
118 n = s->Len();
119 iter = 0;
120 }
121
122 // True if we're done iterating, false if not.
123 bool IsDoneIterating() const { return iter >= n; }
124
125 // Indicates that the current iteration is finished.
126 void IterFinished() { ++iter; }
127
128 // Counter of where we are in the iteration.
129 zeek_uint_t iter; // initialized to 0 at start of loop
130 zeek_uint_t n; // we loop from 0 ... n-1
131
132 // The low-level value we're iterating over.

134 const String* s;
