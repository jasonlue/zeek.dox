144/* A simple buffer for reassembling the fragments that RPC-over-TCP
145 * uses. Only needed by RPC_Contents.
146
147 * However, RPC messages can be quite large. As a first step, we only
148 * extract and analyzer the first part of an RPC message and skip
149 * over the rest.
150 *
151 * We specify:
152 * maxsize: the number of bytes we want to copy into the buffer to analyze.
153 * expected: the total number of bytes in the RPC message. Can be
154 * quite large. We will be "skipping over" expected-maxsize bytes.
155 *
156 * We can extend "expected" (by calling AddToExpected()), but maxsize is
157 * fixed.
158 *
159 * TODO: grow buffer dynamically
160 */
161class RPC_Reasm_Buffer
162 {
163public:
164 RPC_Reasm_Buffer()
165 {
166 maxsize = expected = 0;
167 fill = processed = 0;
168 buf = nullptr;
169 };
170
171 ~RPC_Reasm_Buffer()
172 {
173 if ( buf )
174 delete[] buf;
175 }
176
177 void Init(int64_t arg_maxsize, int64_t arg_expected);
178
179 const u_char* GetBuf() { return buf; } // Pointer to the buffer
180 int64_t GetFill() { return fill; } // Number of bytes in buf
181 int64_t GetSkipped() { return processed - fill; } // How many bytes did we skipped?
182 int64_t GetExpected() { return expected; } // How many bytes are we expecting?
183 int64_t GetProcessed() { return processed; } // How many bytes are we expecting?
184
185 // Expand expected by delta bytes. Returns false if the number of
186 // expected bytes exceeds maxsize (which means that we will truncate
187 // the message).
188 bool AddToExpected(int64_t delta)
189 {
190 expected += delta;
191 return ! (expected > maxsize);
192 }
193
194 // Consume a chunk of input data (pointed to by data, up len in
195 // size). data and len will be adjusted accordingly. Returns true if
196 // "expected" bytes have been processed, i.e., returns true when we
197 // don't expect any more data.
198 bool ConsumeChunk(const u_char*& data, int& len);
199
200protected:
201 int64_t fill; // how many bytes we currently have in the buffer
202 int64_t maxsize; // maximum buffer size we want to allocate
203 int64_t processed; // number of bytes we have processed so far
204 int64_t expected; // number of input bytes we expect
205 u_char* buf;
206 };
