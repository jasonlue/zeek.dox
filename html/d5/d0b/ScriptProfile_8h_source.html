18// Base class for tracking an instance of profile information. The instance
19// can be a single function body (equivalently, event handler or hook body),
20// or an aggregate that includes multiple function bodies.

22class ScriptProfileStats

49 // Used to count instances in an aggregate.

51
52 // Fold into this profile another profile. Second argument controls
53 // whether the folding should include increasing the number of calls.
54 void AddIn(const ScriptProfileStats* eps, bool bump_num_calls = true)
55 {
56 if ( bump_num_calls )
57 ncalls += eps->NumCalls();
58
59 CPU_time += eps->CPUTime();
60 memory += eps->Memory();
61 }
62
63 // Accumulate a single instance of CPU & memory usage.
64 void AddIn(double delta_CPU_time, uint64_t delta_memory)
65 {
66 CPU_time += delta_CPU_time;
67 memory += delta_memory;
68 }
69
70 // Directly specify the total CPU & memory usage.
71 void SetStats(double arg_CPU_time, uint64_t arg_memory)
72 {
73 CPU_time = arg_CPU_time;
74 memory = arg_memory;
75 }
76
77 // Track that the instance has had another call.
78 void NewCall() { ++ncalls; }
79
80private:
81 std::string name;
82 int ninstances = 0;
83 int ncalls = 0;
84 double CPU_time = 0.0;

90class ScriptProfile : public ScriptProfileStats
91 {
92public:
93 ScriptProfile(const Func* _func, const detail::StmtPtr& body)
94 : ScriptProfileStats(_func->Name())
95 {
96 func = {NewRef{}, const_cast<Func*>(_func)};
97 is_BiF = body == nullptr;
98
99 if ( is_BiF )
100 loc = *func->GetLocationInfo();
101 else
102 loc = *body->GetLocationInfo();
103 }
104
105 // Constructor used for the special case of non-script acocunting.
106 ScriptProfile() : ScriptProfileStats("non-scripts")
107 {
108 func = nullptr;
109 is_BiF = false;
110 }
111
112 // Called to register the beginning/end of an execution instance.
113 void StartActivation();
114 void EndActivation();
115
116 // Called when a child instance finishes.
117 void ChildFinished(const ScriptProfile* child);
118
119 bool IsBiF() const { return is_BiF; }
120 double DeltaCPUTime() const { return delta_stats.CPUTime(); }
121 uint64_t DeltaMemory() const { return delta_stats.Memory(); }
122
123 // Write the profile to the given file.
124 void Report(FILE* f) const;
125
126private:
127 // We store "func" as a FuncPtr to ensure it sticks around when
128 // it would otherwise be ephemeral (i.e., for lambdas).
129 FuncPtr func;
130 bool is_BiF;
131 detail::Location loc;
132
133 // Profile associated with child instances (functions or hooks
134 // that this instance calls - does not include events that this
135 // instance generates).
136 ScriptProfileStats child_stats;
137
138 // These are ephemeral, only relevant between Start and End activations.
139 ScriptProfileStats start_stats;

141 // Defined for the last activation period.
