183template <typename T>
184T& require_data_type(broker::data& d, zeek::TypeTag tag, zeek::detail::Frame* f)
185 {
186 auto ptr = broker::get_if<T>(&d);
187 if ( ! ptr )
188 zeek::reporter->RuntimeError(f->GetCallLocation(), "data is of type '%s' not of type '%s'",
189 visit(type_name_getter{tag}, d), zeek::type_name(tag));
190
191 return *ptr;
192 }
