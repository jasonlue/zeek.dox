124 ReaderInfo(const ReaderInfo& other)
125 {
126 source = other.source ? util::copy_string(other.source) : nullptr;
127 name = other.name ? util::copy_string(other.name) : nullptr;
128 mode = other.mode;
129
130 for ( config_map::const_iterator i = other.config.begin(); i != other.config.end();
131 i++ )
132 config.insert(
133 std::make_pair(util::copy_string(i->first), util::copy_string(i->second)));
134 }
135
136 ~ReaderInfo()
137 {
138 delete[] source;
139 delete[] name;
140
141 for ( config_map::iterator i = config.begin(); i != config.end(); i++ )
142 {
143 delete[] i->first;
144 delete[] i->second;
145 }
146 }
