14// Abstract class for capturing a single difference between two script-level

18public:

21
22 // Return a string that performs the update operation, expressed
23 // as Zeek scripting. Does not include a terminating semicolon.
24 virtual std::string Generate(ValTraceMgr* vtm) const;
25
26 // Whether the generated string needs the affected value to
27 // explicitly appear on the left-hand-side. Note that this

42// Tracks the elements of a value as seen at a given point in execution.
43// For non-aggregates, this is simply the Val object, but for aggregates
44// it is (recursively) each of the sub-elements, in a manner that can then
45// be readily compared against future instances.
46class ValTrace
47 {
48public:
49 ValTrace(const ValPtr& v);
50 ~ValTrace();
51
52 const ValPtr& GetVal() const { return v; }
53 const TypePtr& GetType() const { return t; }
54 const auto& GetElems() const { return elems; }
55
56 // Returns true if this trace and the given one represent the
57 // same underlying value. Can involve subelement-by-subelement
58 // (recursive) comparisons.
59 bool operator==(const ValTrace& vt) const;
60 bool operator!=(const ValTrace& vt) const { return ! ((*this) == vt); }
61
62 // Computes the deltas between a previous ValTrace and this one.
63 // If "prev" is nil then we're creating this value from scratch
64 // (though if it's an aggregate, we may reuse existing values
65 // for some of its components).
66 //
67 // Returns the accumulated differences in "deltas". If on return
68 // nothing was added to "deltas" then the two ValTrace's are equivalent
69 // (no changes between them).
70 void ComputeDelta(const ValTrace* prev, DeltaVector& deltas) const;
71
72private:
73 // Methods for tracing different types of aggregate values.
74 void TraceList(const ListValPtr& lv);
75 void TraceRecord(const RecordValPtr& rv);
76 void TraceTable(const TableValPtr& tv);
77 void TraceVector(const VectorValPtr& vv);
78
79 // Predicates for comparing different types of aggregates for equality.
80 bool SameList(const ValTrace& vt) const;
81 bool SameRecord(const ValTrace& vt) const;
82 bool SameTable(const ValTrace& vt) const;
83 bool SameVector(const ValTrace& vt) const;
84
85 // Helper function that knows about the internal vector-of-subelements
86 // we use for aggregates.
87 bool SameElems(const ValTrace& vt) const;
88
89 // True if this value is a singleton and it's the same value as
90 // represented in "vt".
91 bool SameSingleton(const ValTrace& vt) const;
92
93 // Add to "deltas" the differences needed to turn a previous instance
94 // of the given type of aggregate to the current instance.
95 void ComputeRecordDelta(const ValTrace* prev, DeltaVector& deltas) const;
96 void ComputeTableDelta(const ValTrace* prev, DeltaVector& deltas) const;
97 void ComputeVectorDelta(const ValTrace* prev, DeltaVector& deltas) const;
98
99 // Holds sub-elements for aggregates.
100 std::vector<std::shared_ptr<ValTrace>> elems;
101
102 // A parallel vector used for the yield values of tables.
103 std::vector<std::shared_ptr<ValTrace>> elems2;
104
105 ValPtr v;
106 TypePtr t; // v's type, for convenience
107 };

304// Tracks a single event.
305class EventTrace
306 {
307public:
308 // Constructed in terms of the associated script function, "network
309 // time" when the event occurred, and the position of this event
310 // within all of those being traced.
311 EventTrace(const ScriptFunc* _ev, double _nt, int event_num);
312
313 // Sets a string representation of the arguments (values) being
314 // passed to the event.
315 void SetArgs(std::string _args) { args = std::move(_args); }
316
317 // Adds to the trace an update for the given value.
318 void AddDelta(ValPtr val, std::string rhs, bool needs_lhs, bool is_first_def)
319 {
320 auto& d = is_post ? post_deltas : deltas;
321 d.emplace_back(DeltaGen(val, rhs, needs_lhs, is_first_def));
322 }

324 // Initially we analyze events pre-execution. When this flag
325 // is set, we switch to instead analyzing post-execution. The
326 // difference allows us to annotate the output with "# from script"
327 // comments that flag changes created by script execution rather
328 // than event engine activity.
329 void SetDoingPost() { is_post = true; }
330
331 const char* GetName() const { return name.c_str(); }
332
333 // Generates an internal event handler that sets up the values
334 // associated with the traced event, followed by queueing the traced
335 // event, and then queueing the successor internal event handler,
336 // if any.
337 //
338 // "predecessor", if non-nil, gives the event that came just before
339 // this one (used for "# from script" annotations"). "successor",
340 // if not empty, gives the name of the successor internal event.
341 void Generate(FILE* f, ValTraceMgr& vtm, const EventTrace* predecessor,
342 std::string successor) const;
343
344private:
345 // "dvec" is either just our deltas, or the "post_deltas" of our
346 // predecessor plus our deltas.
347 void Generate(FILE* f, ValTraceMgr& vtm, const DeltaGenVec& dvec, std::string successor,
348 int num_pre = 0) const;
349
350 const ScriptFunc* ev;
351 double nt;
352 bool is_post = false;
353
354 // The deltas needed to construct the values associated with this
355 // event prior to its execution.
356 DeltaGenVec deltas;
357
358 // The deltas capturing any changes to the original values as induced
359 // by executing its event handlers.
360 DeltaGenVec post_deltas;
361
362 // The event's name and a string representation of its arguments.
363 std::string name;
