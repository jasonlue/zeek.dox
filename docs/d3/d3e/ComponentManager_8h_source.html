157template <class C>
158ComponentManager<C>::ComponentManager(const std::string& module, const std::string& local_id,
159 const std::string& parent_module)
160 : module(module), parent_module(parent_module)
161 {
162 tag_enum_type = make_intrusive<EnumType>(module + "::" + local_id);
163 auto id = zeek::detail::install_ID(local_id.c_str(), module.c_str(), true, true);
164 zeek::detail::add_type(id.get(), tag_enum_type, nullptr);
165 zeek::detail::zeekygen_mgr->Identifier(std::move(id));
166
167 if ( ! parent_module.empty() )
168 {
169 // check to see if the parent module's type has been created already
170 id = zeek::detail::lookup_ID(local_id.c_str(), parent_module.c_str(), false, true, false);
171 if ( id != zeek::detail::ID::nil )
172 {
173 parent_tag_enum_type = id->GetType<EnumType>();
174 }
175 else
176 {
177 parent_tag_enum_type = make_intrusive<EnumType>(parent_module + "::" + local_id);
178 id = zeek::detail::install_ID(local_id.c_str(), parent_module.c_str(), true, true);
179 zeek::detail::add_type(id.get(), parent_tag_enum_type, nullptr);
180 zeek::detail::zeekygen_mgr->Identifier(std::move(id));
181 }
182 }

295template <class C>
296void ComponentManager<C>::RegisterComponent(C* component, const std::string& prefix)
297 {
298 std::string cname = component->CanonicalName();
299
300 if ( Lookup(cname) )
301 reporter->FatalError("Component '%s::%s' defined more than once", module.c_str(),
302 cname.c_str());
303
304 DBG_LOG(DBG_PLUGINS, "Registering component %s (tag %s)", component->Name().c_str(),
305 component->Tag().AsString().c_str());
306
307 components_by_name.insert(std::make_pair(cname, component));
308 components_by_tag.insert(std::make_pair(component->Tag(), component));
309 components_by_val.insert(std::make_pair(component->Tag().AsVal()->InternalInt(), component));
310
311 // Install an identfier for enum value
312 std::string id = util::fmt("%s%s", prefix.c_str(), cname.c_str());
313 tag_enum_type->AddName(module, id.c_str(), component->Tag().AsVal()->InternalInt(), true,
314 nullptr);
315
316 if ( parent_tag_enum_type )
317 {
318 std::string parent_id = util::fmt("%s_%s", util::strtoupper(module).c_str(), id.c_str());
319 parent_tag_enum_type->AddName(parent_module, parent_id.c_str(),
320 component->Tag().AsVal()->InternalInt(), true, nullptr);
321 }
322 }
