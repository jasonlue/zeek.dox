533template <class T> std::list<T*> Manager::Components() const
534 {
535 std::list<T*> result;
536
537 for ( plugin_list::const_iterator p = ActivePluginsInternal()->begin();
538 p != ActivePluginsInternal()->end(); p++ )
539 {
540 component_list components = (*p)->Components();
541
542 for ( component_list::const_iterator c = components.begin(); c != components.end(); c++ )
543 {
544 T* t = dynamic_cast<T*>(*c);
545
546 if ( t )
547 result.push_back(t);
548 }
549 }
550
551 return result;
552 }
