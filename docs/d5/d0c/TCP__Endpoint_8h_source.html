24enum EndpointState
25 {
26 TCP_ENDPOINT_INACTIVE, // no SYN (or other packets) seen for this side
27 TCP_ENDPOINT_SYN_SENT, // SYN seen, but no ack
28 TCP_ENDPOINT_SYN_ACK_SENT, // SYN ack seen, no initial SYN
29 TCP_ENDPOINT_PARTIAL, // data seen, but no SYN
30 TCP_ENDPOINT_ESTABLISHED, // SYN ack seen (implicit for SYN
31 // sent by responder)
32 TCP_ENDPOINT_CLOSED, // FIN seen
33 TCP_ENDPOINT_RESET // RST seen
34 };
