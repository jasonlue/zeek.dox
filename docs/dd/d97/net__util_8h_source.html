209// True if sequence # a is between b and c (b <= a <= c). It must be true
210// that b <= c in the sequence space.

286inline double ntohd(double d)
287 {
288 assert(sizeof(d) == 8);
289
290 double tmp;
291 char* src = (char*)&d;
292 char* dst = (char*)&tmp;
293
294 dst[0] = src[7];
295 dst[1] = src[6];
296 dst[2] = src[5];
297 dst[3] = src[4];
298 dst[4] = src[3];
299 dst[5] = src[2];
300 dst[6] = src[1];
301 dst[7] = src[0];
302
303 return tmp;
304 }
305
306inline double htond(double d)
307 {
308 return ntohd(d);
309 }

311inline float ntohf(float f)
312 {
313 assert(sizeof(f) == 4);
314
315 float tmp;
316 char* src = (char*)&f;
317 char* dst = (char*)&tmp;
318
319 dst[0] = src[3];
320 dst[1] = src[2];
321 dst[2] = src[1];
322 dst[3] = src[0];
323
324 return tmp;
325 }

348 c = x.c[2];
349 x.c[2] = x.c[5];
350 x.c[5] = c;
351 c = x.c[3];
352 x.c[3] = x.c[4];
353 x.c[4] = c;
