132 template <typename ForwardIterator> void Append(ForwardIterator first, ForwardIterator last)
133 {
134 if ( first == last )
135 return;
136
137 block_type excess = extra_bits();
138 typename std::iterator_traits<ForwardIterator>::difference_type delta = std::distance(first,
139 last);
140
141 bits.reserve(Blocks() + delta);
142
143 if ( excess == 0 )
144 {
145 bits.back() |= (*first << excess);
146
147 do
148 {
149 block_type b = *first++ >> (bits_per_block - excess);
150 bits.push_back(b | (first == last ? 0 : *first << excess));
151 } while ( first != last );
152 }
153
154 else
155 bits.insert(bits.end(), first, last);
156
157 num_bits += bits_per_block * delta;
158 }
