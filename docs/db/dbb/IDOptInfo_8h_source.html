24// This class tracks a single region during which an identifier has
25// a consistent state of definition, meaning either it's (1) defined
26// as of its value after a specific statement, (2) might-or-might-not
27// be defined, or (3) definitely not defined.
28
29class IDDefRegion
30 {
31public:
32 IDDefRegion(const Stmt* s, bool maybe, int def);
33 IDDefRegion(int stmt_num, int level, bool maybe, int def);
34 IDDefRegion(const Stmt* s, const IDDefRegion& ur);
35
36 void Init(bool maybe, int def)
37 {
38 if ( def != NO_DEF )
39 maybe_defined = true;
40 else
41 maybe_defined = maybe;
42
43 defined = def;
44 }
45
46 // Returns the starting point of the region, i.e., the number
47 // of the statement *after* which executing this region begins.
48 int StartsAfter() const { return start_stmt; }
49
50 // Returns or sets the ending point of the region, i.e., the
51 // last statement for which this region applies (including executing
52 // that statement). A value of NO_DEF means that the region
53 // continues indefinitely, i.e., we haven't yet encountered its end.
54 int EndsAfter() const { return end_stmt; }
55 void SetEndsAfter(int _end_stmt) { end_stmt = _end_stmt; }
56
57 // The confluence nesting level associated with the region. Other
58 // regions that overlap take precedence if they have a higher
59 // (= more inner) block level.
60 int BlockLevel() const { return block_level; }
61
62 // True if in the region the identifer could be defined.
63 bool MaybeDefined() const { return maybe_defined; }
64
65 // Returns (or sets) the statement after which the identifer is
66 // (definitely) defined, or NO_DEF if it doesn't have a definite
67 // point of definition.
68 int DefinedAfter() const { return defined; }
69 void UpdateDefinedAfter(int _defined) { defined = _defined; }
70
71 // Returns (or sets) the expression used to define the identifier,
72 // if any. Note that an identifier can be definitely defined
73 // (i.e., DefinedAfter() returns a statement number, not NO_DEF)
74 // but not have an associated expression, if the point-of-definition
75 // is the end of a confluence block.
76 const ExprPtr& DefExprAfter() const { return def_expr; }
77 void SetDefExpr(ExprPtr e) { def_expr = e; }
78
79 // Used for debugging.
80 void Dump() const;
81
82protected:
83 // Number of the statement for which this region applies *after*
84 // its execution.
85 int start_stmt;
86
87 // Number of the statement that this region applies to, *after*
88 // its execution.
89 int end_stmt = NO_DEF; // means the region hasn't ended yet
90
91 // Degree of confluence nesting associated with this region.
92 int block_level;
93
94 // Identifier could be defined in this region.
95 bool maybe_defined;
96
97 // If not NO_DEF, then the statement number of either the identifier's
98 // definition, or its confluence point if multiple, differing
99 // definitions come together.
100 int defined;
101
102 // The expression used to define the identifier in this region.
103 // Nil if either it's ambiguous (due to confluence), or the
104 // identifier isn't guaranteed to be defined.
105 ExprPtr def_expr;
106 };

138 // Used to track expressions employed when explicitly initializing
139 // the (global) identifier. These are needed by compile-to-C++ script

161 // Whether the identifier is a temporary variable. Temporaries
162 // are guaranteed to have exactly one point of definition.

166 // Called when the identifier is defined via execution of the
167 // given statement, with an assignment to the expression 'e'
168 // (only non-nil for simple direct assignments). "conf_blocks"
169 // gives the full set of surrounding confluence statements.
170 // It should be processed starting at conf_start (note that

172 void DefinedAfter(const Stmt* s, const ExprPtr& e, const std::vector<const Stmt*>& conf_blocks,

175 // Called upon encountering a "return" statement.
176 void ReturnAt(const Stmt* s);
177
178 // Called when the current region ends with a backwards branch,
179 // possibly across multiple block levels, occurring at "from"
180 // and going into the block "to". If "close_all" is true then
181 // any pending regions at a level inner to "to" should be
182 // closed; if not, just those at "from"'s level.
183 void BranchBackTo(const Stmt* from, const Stmt* to, bool close_all);
184
185 // Called when the current region ends at statement end_s with a
186 // forwards branch, possibly across multiple block levels, to
187 // the statement that comes right after the execution of "block".
188 // See above re "close_all".
189 void BranchBeyond(const Stmt* end_s, const Stmt* block, bool close_all);
190
191 // Start tracking a confluence block that begins with the body
192 // of s (not s itself).
193 void StartConfluenceBlock(const Stmt* s);
194
195 // Finish tracking confluence; s is the last point of execution
196 // prior to leaving a block. If no_orig_flow is true, then
197 // the region for 's' itself does not continue to the end of
198 // the block.
199 void ConfluenceBlockEndsAfter(const Stmt* s, bool no_orig_flow);
200
201 // All of these regard the identifer's state just *prior* to

247 // Expressions used to initialize the identifier, for use by
248 // the scripts-to-C++ compiler. We need to track all of them
249 // because it's possible that a global value gets created using
250 // one of the earlier instances rather than the last one.
251 std::vector<ExprPtr> init_exprs;
252
253 // Tracks initializations of globals in the order they're seen.
254 static std::vector<IDInitInfo> global_init_exprs;
255
256 // If non-nil, a constant that this identifier always holds
257 // once initially defined.
258 const ConstExpr* const_expr = nullptr;
259
260 // The different usage regions associated with the identifier.
261 // These are constructed such that they're always with non-decreasing
262 // starting statements.
263 std::vector<IDDefRegion> usage_regions;
264
265 // A type for collecting the indices of usage_regions that will
266 // all have confluence together at one point. Used to track
267 // things like "break" statements that jump out of loops or

269 using ConfluenceSet = std::set<int>;
270
271 // Maps loops/switches/catch-returns to their associated

273 std::map<const Stmt*, ConfluenceSet> pending_confluences;
274
275 // A stack of active confluence statements, so we can always find

278
279 // Parallel vector that tracks whether, upon creating the

281 // going beyond it. If so, then we can ignore no_orig_flow when
282 // ending the block, because in fact there *was* original flow.
283 std::vector<bool> block_has_orig_flow;
284
285 // Whether the identifier is a temporary variable.
286 bool is_temp = false;
287
288 // Associated identifier, to enable tracking of initialization
289 // expressions for globals (for C++ compilation), and for debugging
290 // output.
291 const ID* my_id;
292
293 // Only needed for debugging purposes.
294 bool tracing = false;
295
296 // Track whether we've already generated usage errors.
297 bool did_undefined_warning = false;
298 bool did_possibly_undefined_warning = false;
299 };
