54// This needs to remain a std::list because of the usage of iterators in the
55// Analyzer::Forward methods. These methods have the chance to loop back
56// into the same analyzer in the case of tunnels. If the recursive call adds
57// to the children list, it can invalidate iterators in the outer call,
58// causing a crash.
59using analyzer_list = std::list<Analyzer*>;
