62 void ChangeNext(uint8_t next_type)
63 {
64 switch ( type )
65 {
66 case IPPROTO_IPV6:
67 ((ip6_hdr*)data)->ip6_nxt = next_type;
68 break;
69 case IPPROTO_HOPOPTS:
70 case IPPROTO_DSTOPTS:
71 case IPPROTO_ROUTING:
72 case IPPROTO_FRAGMENT:
73 case IPPROTO_AH:
74 case IPPROTO_MOBILITY:
75 ((ip6_ext*)data)->ip6e_nxt = next_type;
76 break;
77 case IPPROTO_ESP:
78 default:
79 break;
80 }
81 }

89 uint8_t NextHdr() const
90 {
91 switch ( type )
92 {
93 case IPPROTO_IPV6:
94 return ((ip6_hdr*)data)->ip6_nxt;
95 case IPPROTO_HOPOPTS:
96 case IPPROTO_DSTOPTS:
97 case IPPROTO_ROUTING:
98 case IPPROTO_FRAGMENT:
99 case IPPROTO_AH:
100 case IPPROTO_MOBILITY:
101 return ((ip6_ext*)data)->ip6e_nxt;
102 case IPPROTO_ESP:
103 default:
104 return IPPROTO_NONE;
105 }
106 }

111 uint16_t Length() const
112 {
113 switch ( type )
114 {
115 case IPPROTO_IPV6:
116 return 40;
117 case IPPROTO_HOPOPTS:
118 case IPPROTO_DSTOPTS:
119 case IPPROTO_ROUTING:
120 case IPPROTO_MOBILITY:
121 return 8 + 8 * ((ip6_ext*)data)->ip6e_len;
122 case IPPROTO_FRAGMENT:
123 return 8;
124 case IPPROTO_AH:
125 return 8 + 4 * ((ip6_ext*)data)->ip6e_len;
126 case IPPROTO_ESP:
127 return 8; // encrypted payload begins after 8 bytes
128 default:
129 return 0;
130 }
