68 {
69 // A bit of a hack. We'd like to associate location
70 // information with every object created when parsing,
71 // since for them, the location is generally well-defined.
72 // We could maintain a separate flag that tells us whether
73 // we're inside a parse, but the parser also sets the
74 // location to no_location when it's done, so it makes
75 // sense to just check for that. *However*, start_location
76 // and end_location are maintained as their own objects
77 // rather than pointers or references, so we can't directly
78 // check them for equality with no_location. So instead
79 // we check for whether start_location has a line number
80 // of 0, which should only happen if it's been assigned
81 // to no_location (or hasn't been initialized at all).
82 location = nullptr;
83 if ( detail::start_location.first_line != 0 )
84 SetLocationInfo(&detail::start_location, &detail::end_location);
85 }
